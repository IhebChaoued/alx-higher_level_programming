# Import the MyList class
MyList = __import__('1-my_list').MyList

# Test Case 1: Sorting an unsorted list
my_list = MyList([4, 2, 1, 3, 5])
my_list.print_sorted()  # This should print [1, 2, 3, 4, 5]

# Test Case 2: Sorting an already sorted list
my_list = MyList([1, 2, 3, 4, 5])
my_list.print_sorted()  # This should print [1, 2, 3, 4, 5]

# Test Case 3: Sorting an empty list

my_list.print_sorted()  # This should print []
